1function dfr = rffe(df,ft,cas,cau,cef)
 2% RFFE: Radio Freq Front End: AGC's, Freq Compensates, & RCRF's Input Data
 3% Function Arguments: 
 4% cas: Constant AGC Step Size (fixed at 1.0)
 5% cau: Constant AGC Update Rate (fixed at usrpFrameLength)
 6% cef: Constant Frequency Offset Estimation Freq Resolution (~4.0 Hz)
 7% df:  This Data Frame: Raw data taken from USRP Rx Buffer (length usrpFrameLength)
 8% dfr: This Data Frame Recovered: Data after AGC, FOC & RCRF (length halfUsrpFrameLength)
 9% ft:  Flag Terminal: If true, specifies to release all System objects
10
11% setting global variables
12global spreadFactor usrpFrameLength halfUsrpFrameLength samplingRate
13
14% ha: Handle to Automatic Gain Control (AGC) System object
15persistent ha;
16% hd: Handle to FIR Decimator System object
17persistent hd;
18% he: Handle to Coarse Frequency Estimator (CFE) System object
19persistent he;
20% hf: Handle to Frequency Offset Compensator (FOC) System object
21persistent hf;
22% hrr: Handle to Raised Cosine Receive Filter (RCRF) System object
23persistent hrr;
24% Initialize Persistent Data: Only on first call to RFFE()
25if isempty(ha)
26%     ha = comm.AGC('DetectorMethod','Rectifier','LoopMethod','Logarithmic', ...
27%         'MaximumGain',30,'StepSize',cas,'UpdatePeriod',cau);
28    ha = comm.AGC('MaxPowerGain',30,'AdaptationStepSize',cas);
29end % END IF ISEMPTY(HA)
30if isempty(hd)
31    hd = dsp.FIRDecimator('DecimationFactor',(spreadFactor*2));
32end % END IF ISEMPTY(HD)
33if isempty(he)
34    he = comm.PSKCoarseFrequencyEstimator('Algorithm','FFT-based', ...
35        'FrequencyResolution',cef,'ModulationOrder',2,'SampleRate',(samplingRate/(spreadFactor*2)));
36end % END IF ISEMPTY(HC)
37if isempty(hf)
38    hf = comm.PhaseFrequencyOffset('PhaseOffset',0, ...
39        'FrequencyOffsetSource','Input port','SampleRate',samplingRate);
40end % END IF ISEMPTY(HF)
41if isempty(hrr)
42    hrr = comm.RaisedCosineReceiveFilter('DecimationFactor',2,...
43        'DecimationOffset',0,'FilterSpanInSymbols',8,'Gain',1,...
44        'InputSamplesPerSymbol',2,'RolloffFactor',0.3,'Shape','Square root');
45end % IF ISEMPTY(HRR)
46% Local data preallocation
47ddf = complex(zeros(64,1)); %#ok<NASGU>
48dfr = complex(zeros(halfUsrpFrameLength,1));
49ofs = double(0.0); %#ok<NASGU>
50tmp = complex(zeros(usrpFrameLength,1)); %#ok<NASGU>
51
52if ft
53    % Release System objects: only on final call to RFFE()
54    release(ha);
55    release(hd);
56    release(he);
57    release(hf);
58    release(hrr);
59    %clear('ha','he','hf','hrr'); %Not supported for code generation
60else
61    % Automatic Gain Control (AGC)
62    tmp = step(ha,df);
63    % FIR Decimation
64    ddf = step(hd,tmp);
65    % Coarse Frequency Offset Estimation (CFE)
66    ofs = step(he,ddf);
67    % Frequency Offset Compensation (FOC)
68    tmp = step(hf,tmp,-ofs);
69    % Raised Cosine Receive Filtration (RCRF)
70    dfr = step(hrr,tmp);
71end % IF FT
72end % FUNCTION RFFE
