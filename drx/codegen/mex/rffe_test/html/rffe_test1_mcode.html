1function dfr = rffe_test(df,ft,cas,cau,cef)
 2% RFFE: Radio Freq Front End: AGC's, Freq Compensates, & RCRF's Input Data
 3% Function Arguments: 
 4% cas: Constant AGC Step Size (fixed at 1.0)
 5% cau: Constant AGC Update Rate (fixed at usrpFrameLength)
 6% cef: Constant Frequency Offset Estimation Freq Resolution (~4.0 Hz)
 7% df:  This Data Frame: Raw data taken from USRP Rx Buffer (length usrpFrameLength)
 8% dfr: This Data Frame Recovered: Data after AGC, FOC & RCRF (length halfUsrpFrameLength)
 9% ft:  Flag Terminal: If true, specifies to release all System objects
10
11% setting global variables
12spreadFactor = 11;
13usrpFrameLength = 1408;
14halfUsrpFrameLength = 704;
15samplingRate = 2e5;
16
17% ha: Handle to Automatic Gain Control (AGC) System object
18persistent ha;
19% hd: Handle to FIR Decimator System object
20persistent hd;
21% he: Handle to Coarse Frequency Estimator (CFE) System object
22persistent he;
23% hf: Handle to Frequency Offset Compensator (FOC) System object
24persistent hf;
25% hrr: Handle to Raised Cosine Receive Filter (RCRF) System object
26persistent hrr;
27% Initialize Persistent Data: Only on first call to RFFE()
28if isempty(ha)
29%     ha = comm.AGC('DetectorMethod','Rectifier','LoopMethod','Logarithmic', ...
30%         'MaximumGain',30,'StepSize',cas,'UpdatePeriod',cau);
31    ha = comm.AGC('MaxPowerGain',30,'AdaptationStepSize',cas);
32end % END IF ISEMPTY(HA)
33if isempty(hd)
34    hd = dsp.FIRDecimator('DecimationFactor',(spreadFactor*2));
35end % END IF ISEMPTY(HD)
36if isempty(he)
37    he = comm.PSKCoarseFrequencyEstimator('Algorithm','FFT-based', ...
38        'FrequencyResolution',cef,'ModulationOrder',2,'SampleRate',(samplingRate/(spreadFactor*2)));
39end % END IF ISEMPTY(HC)
40if isempty(hf)
41    hf = comm.PhaseFrequencyOffset('PhaseOffset',0, ...
42        'FrequencyOffsetSource','Input port','SampleRate',samplingRate);
43end % END IF ISEMPTY(HF)
44if isempty(hrr)
45    hrr = comm.RaisedCosineReceiveFilter('DecimationFactor',2,...
46        'DecimationOffset',0,'FilterSpanInSymbols',8,'Gain',1,...
47        'InputSamplesPerSymbol',2,'RolloffFactor',0.3,'Shape','Square root');
48end % IF ISEMPTY(HRR)
49% Local data preallocation
50ddf = complex(zeros(64,1)); %#ok<NASGU>
51dfr = complex(zeros(halfUsrpFrameLength,1));
52ofs = double(0.0); %#ok<NASGU>
53tmp = complex(zeros(usrpFrameLength,1)); %#ok<NASGU>
54
55if ft
56    % Release System objects: only on final call to RFFE()
57    release(ha);
58    release(hd);
59    release(he);
60    release(hf);
61    release(hrr);
62    %clear('ha','he','hf','hrr'); %Not supported for code generation
63else
64    % Automatic Gain Control (AGC)
65    tmp = step(ha,df);
66    % FIR Decimation
67    ddf = step(hd,tmp);
68    % Coarse Frequency Offset Estimation (CFE)
69    ofs = step(he,ddf);
70    % Frequency Offset Compensation (FOC)
71    tmp = step(hf,tmp,-ofs);
72    % Raised Cosine Receive Filtration (RCRF)
73    dfr = step(hrr,tmp);
74end % IF FT
75end % FUNCTION RFFE
