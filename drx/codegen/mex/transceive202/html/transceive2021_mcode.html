1function [dr,ns] = transceive202(d2s,ft,txGain, rxGain, centerFreqTx, centerFreqRx, intFactor, decFactor,samplingRate)
 2persistent hrx htx; 
 3dr = complex(zeros(1408,1));
 4ns = uint32(0);
 5if isempty(htx)
 6    htx = comm.SDRuTransmitter('CenterFrequency',centerFreqTx,'Gain',txGain, ...
 7        'InterpolationFactor',intFactor,'IPAddress','192.168.20.2', ...
 8        'LocalOscillatorOffset',0);
 9end
10if isempty(hrx)
11    hrx = comm.SDRuReceiver('CenterFrequency',centerFreqRx, ...
12        'DecimationFactor',decFactor,'FrameLength',1408,'Gain',rxGain, ...
13        'IPAddress','192.168.20.2','OutputDataType','double', ...
14        'SampleRate',samplingRate);
15end
16if (ft)
17    release(hrx);
18    release(htx);
19else
20    step(htx,d2s);
21    while (ns < uint32(1))
22        [dr,ns] = step(hrx); 
23    end    
24end
25return;
26end
